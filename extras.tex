\section{АЛГЕБРА}

\begin{dfn}\label{eq-class}
    Пусть $X$ -- множество, $\sim$ -- отношение эквиваленции на $X$.
    \textit{Классом эквивалентности по отношению $\sim$ элемента $x\in X$}
    назовем множество
    $$
    X_x = \set{x'\in X | x \sim x'}
    $$
\end{dfn}

\begin{dfn}
    Множество, элементами которого являются классы эквивалентности (из \ref{eq-class})
    обозначим $X/\sim$ и назовем \textit{фактормножеством}.
\end{dfn}

\begin{dfn}
    Отображение $f: X \To X/\sim$ называется \textit{естественным}.
\end{dfn}

\begin{dfn}
    Множество всех отображений $Y$ в $X$ называется $X^Y$
\end{dfn}

\begin{dfn}
    Пусть $L$ -- кольцо. Множество $L^*$ называется
    \textit{множеством обратимых элементов кольца $L$} и определяется следующим образом
    $$
    L^* = \set{l\in L|\exists l' \in L: ll' = l'l = 1_L}
    $$
\end{dfn}

\begin{dfn}
    Пусть $R$ -- кольцо. Левый модуль $M$ кольца $R$ или левый $R$-модуль $M$,
    если для любых $a,b \in R$ и $x,y \in M$ выполняется следующее:
    \begin{enumerate}
        \item $(a + b)x = ax + bx$;
        \item $a(x + y) = ax + ay$.
    \end{enumerate}
\end{dfn}

Аналогичным образом определяется правый $R$-модуль.
Если модуль $M$ является одновременно левым и правым, то будем называть его просто $R$-модуль.

\begin{dfn}
    Пусть $A,B$ -- кольца. Отображение $f: A \To B$ называется \textit{морфизмом колец}, если:
    \begin{enumerate}
        \item $f(1) = 1$;
        \item $f(x + y) = f(x) + f(y)$;
        \item $f(xy) = f(x)f(y)$.
    \end{enumerate}
\end{dfn}

\begin{dfn}\cite[стр. 119]{Lang}
    Пусть $M, M'$ -- модули кольца $R$. Морфизмом модулей будем называть отображение
    такое $f: M \To M'$, что для любых $a,b \in R$ и $x,y \in M$ выполняется:
    $$
    f(ax + by) = af(x) + bf(y)
    $$
\end{dfn}

\begin{dfn}
    Инъективный морфизм $f: X \To Y$ называется \textit{мономорфизмом}.
    Обозначается $f: X \rightarrowtail Y$.
\end{dfn}

\begin{dfn}
    Сюръективный морфизм $f: X \To Y$ называется \textit{эпиморфизмом}.
    Обозначается $f: X \twoheadrightarrow Y$.
\end{dfn}

\section{ИДЕАЛЫ}

\begin{dfn}
    \textit{Идеалом кольца} $R$ называется произвольное подмножество $I \subseteq R$ такое, что
    \begin{enumerate}
        \item $I \neq \0$;
        \item Если $a,b \in I$, то $a+b\in I$ (замкнутость по сложению);
        \item Если $a \in I, b\in R$, то $ab \in I$
        (замкнутость по умножению на элементы кольца~$R$).
    \end{enumerate}
\end{dfn}

\begin{example*}
    Пусть $R$ -- кольцо. Подмножества $R,\left\lbrace 0\right\rbrace $ -- идеалы кольца $R$.
    Такие идеалы называются \textit{тривиальными}.
\end{example*}

\begin{dfn}
    Если $a \in R$, то множество
    $$\gen{a} := \set{ab \;|\; b \in R} =: aR = Ra$$
    --- \textit{главный идеал} $R$, порожденный элементом $a$.
\end{dfn}

\begin{example*}
    Пусть $R = \Z$, $m \in \Z$.\\
    $$\gen{m} = m\Z = \set{mt \;|\; t\in \Z}$$
\end{example*}

\begin{dfn}
    Подмножество $S$ кольца $R$ называется \textit{мультипликативным}, если
    \begin{enumerate}
        \item $1 \in S$ (где $1$ -- нейтральный по умножению элемент кольца $R$);
        \item Для любых $a,b \in S: \; ab \in S$.
    \end{enumerate}
\end{dfn}

\begin{dfn}
    Кольцо $R$ называется \textit{целостным}, если оно ненулевое и
    для любых ненулевых $a,b \in R : \; ab \neq 0$.
\end{dfn}

\begin{dfn}
    \textit{Полем} называется кольцо, все ненулевые элементы которого обратимы.
\end{dfn}

\begin{dfn}\cite[стр. 92]{Lang}
    Идеал $I$ кольца $R$ --- \textit{простой}, если $I \neq R$ и всякий раз,
    когда $a,b \in R$ и $ab \in I$, тогда $a \in I$ или $b \in I$.
\end{dfn}

\begin{dfn}\cite[стр. 92]{Lang}
    Идеал $I$ кольца $R$--- \textit{максимальный}, если $I \neq R$ и нет такого идеала $I'$
    отличного от $I$ и содержащего его.
\end{dfn}

\begin{example*}
    Пусть $I$ -- простой идеал кольца $R$, тогда $R \setminus I$ --- мультипликативное подмножество.
\end{example*}

\begin{example*}
    Пусть $f \in R$, $S = \set{f^n|n\in \Z_{\ges 0}}$, тогда $S$ ---
    мультипликативное подмножество $R$.
\end{example*}

\begin{thm}\cite[стр. 92]{Volochkov}
    Идеал $I$ кольца $R$ --- максимальный тогда и только тогда, когда $R/I$ -- поле.
\end{thm}

\begin{thm}\cite[стр. 92]{Lang}
    Любой максимальный идеал --- простой.
\end{thm}

\begin{dfn}
    Кольцо называется \textit{локальным}, если оно имеет ровно один максимальный идеал.
\end{dfn}

\begin{thm}\label{assertion-eq-thm}\cite[стр. 91]{Volochkov}
    Если $I$ -- идеал кольца $R$, то равносильны следующие утверждения:
    \begin{enumerate}
        \item $R \setminus I$ -- мультипликативное подмножество;
        \item $R/I$ -- целостное кольцо.
    \end{enumerate}
\end{thm}

\begin{dfn}
    Идеал $I$ кольца $R$ --- \textit{простой},
    если выполняется хотя бы одно из утверждений теоремы \ref{assertion-eq-thm}.
\end{dfn}

    \textit{Локализацией $A$ по $f$} назовем локализацию $A$ по мультипликативному

    \textit{Аннулятором $S$} назовем следующее множество
    $$
    \operatorname{Ann}(S) = \set{a \in A | aS = \set{0} }
    $$
\end{dfn}

\begin{thm}\cite[стр. 96]{Volochkov}
    Пусть $S$ -- мультипликативное подмножество кольца $A$, $\lambda: A \To L$ -- локализация $A$ по $S$.
    Если $a \in A$, то $\lambda(a) = 0$ $\Leftrightarrow$ $as = 0$ для некоторого $s \in S$.
\end{thm}

\begin{thm}\cite[стр. 97]{Volochkov}
    Пусть $S$ -- мультипликативное подмножество кольца $A$, $\lambda: A \To L$ -- локализация $A$ по $S$.
    Тогда равносильны следующие условия:
    \begin{enumerate}
        \item $L = \{0\}$;
        \item $0 \in S$.
    \end{enumerate}
\end{thm}

\begin{dfn}
    \textit{Радикалом $\sqrt{}I$ идеала} $I$ кольца $R$ назовем множество
    $$
    \{a\in R | (\exists n\in\N ) (a^n \in I)\}
    $$
\end{dfn}

\begin{dfn}
    Радикал нулевого идеала кольца $R$ назовем \textit{нильрадикалом} $R$ и
    обозначим $\operatorname{rad}(R)$. Таким образом,
    $$
    \operatorname{rad}(R) = \{a\in R | (\exists n\in\N ) (a^n = 0)\}
    $$
\end{dfn}

\begin{thm}\cite[стр. 97]{Volochkov}
    Пусть $I$ -- идеал кольца $R$. Тогда $\sqrt{}I$ совпадает
    с пересечением всех простых идеалов содержащих $I$.
\end{thm}

\begin{cor}
    Пусть $R$ -- кольцо. Тогда $\operatorname{rad}(R)$ совпадает
    с пересечением всех простых идеалов кольцо $R$.
\end{cor}

\begin{thm}\cite[стр. 98]{Volochkov}
    Пусть $A$ -- кольцо, $f\in A$, $\lambda: A \To L$ -- локализация $A$ по $f$,
    $a \in A$. Тогда равносильны следующие условия:
    \begin{enumerate}
        \item $\lambda(a) = 0$;
        \item $af^n = 0$, для некоторого $n \in \Z_{\ges 0}$;
        \item $f \in \sqrt{\operatorname{Ann}(a)}$.
    \end{enumerate}
\end{thm}

\begin{thm}\cite[стр. 98]{Volochkov}
    Пусть $S$ -- мультипликативное подмножество кольца $A$,
    $\lambda: A \To L$ -- морфизм колец. Тогда равносильны следующие условия:
    \begin{enumerate}
        \item $\lambda$ -- локализация $A$ по $S$;
        \item $(a)$ $\lambda(S) \subseteq L^*$ \\
            $(b)$ $L = \set{\lambda(a)/\lambda(s)|a\in A, s\in S}$ \\
            $(c)$ $\forall a\in A \; \lambda(a)=0 \Leftrightarrow as=0$ для некоторого $S$.
    \end{enumerate}
\end{thm}

\begin{example*}
    Пусть $B$ -- кольцо, $A$ -- его подкольцо, $S$ -- мультипликативное подмножество кольца $A$,
    $S \subseteq B^*$, $L = \set{a/s|a\in A, s\in S}$,
    $\lambda: A\To L$, $\lambda(a) = a$. Тогда $\lambda$ -- локализация $A$ по $S$.
\end{example*}

\section{МОДУЛИ}

\begin{dfn}
    Модуль $M$ является \textit{прямой суммой} своих подмодулей $M_1$, $M_2$,
    если для любого $m \in M$ найдутся единственный $m_1 \in M_1$ и
    единственный $m_2 \in M_2$, такие, что $m = m_1 + m_2$.
    Выражение <<\textit{$M$ является прямой суммой $M_1$ и $M_2$}>>
    будем записывать $M = M_1 \oplus M_2$.
\end{dfn}

\begin{thm}
    Пусть $M$ модуль над кольцом $R$, тогда равносильны следующие условия:
    \begin{enumerate}
        \item $M = M_1 \oplus M_2$;
        \item $M = M_1 + M_2$ и $M_1 \cap M_2 = \{0\}$.
    \end{enumerate}
\end{thm}

\begin{dfn}
    Подмодуль $S$ модуля $M$ \textit{выделяется прямым слагаемым},
    если найдется такой подмодуль $T$ модуля $M$, что $M = S \oplus T$.
\end{dfn}

\begin{dfn}\label{lin-independence}\cite[стр. 140]{Cohn}
    Пусть $M$ -- $R$-модуль кольца $R$.
    Некоторое семейство $x_1, ... , x_n \in M$ \textit{линейно независимо},
    если для любых $a_i \in R, i\in I$ справедливо $\sum\limits_{i\in I}a_i x_i = 0$ только тогда,
    когда $a_1=...=a_n = 0$.
\end{dfn}

\begin{dfn}
    Семейство $x_1,...,x_n$ из определения \ref{lin-independence}
    будем называть \textit{базисом} $R$-модуля $M$.
\end{dfn}

\begin{dfn}\label{free-module}
    Модуль \textit{свободный}, если имеет базис.
\end{dfn}

\begin{dfn}\label{projective-module}
    Модуль $P$ над кольцом $R$ называется \textit{проективным}, если для
    любого эпиморфизма $f: N \twoheadrightarrow M$ $R$-модулей и
    любого морфизма $g: P \To M$ существует такой морфизм $h: P \To N$,
    что коммутативна диаграмма
    $$
        \begin{tikzcd}
            & N \\
            P & M
            \arrow["g"', from=2-1, to=2-2]
            \arrow["f", two heads, from=1-2, to=2-2]
            \arrow["h", from=2-1, to=1-2]
        \end{tikzcd}
    $$
\end{dfn}

\begin{thm}\label{projectivity-criteria}
    Модуль проективен тогда и только тогда,
    когда он является прямым слагаемым некоторого свободного модуля.
\end{thm}

\begin{thm}
    Проективный модуль над локальным кольцом свободен.
\end{thm}

\begin{thm}
    Пусть $R$ -- коммутативное кольцо, $\mathfrak{p}$ -- простой идеал в $R$. Если
    $M$ -- проективный $R$-модуль, то $M_\mathfrak{p}$ -- свободный $R_\mathfrak{p}$-модуль.
\end{thm}

\begin{dfn}\label{rank-of-projective-module}
    Проективный модуль $M$ над коммутативным кольцом $R$ имеет ранг $r \in \Z_{\ges 0}$,
    если для любого простого идеала $\mathfrak{p}$ кольца $R$
    свободный $R_\mathfrak{p}$-модуль $M_\mathfrak{p}$ имеет (свободный) ранг $r$.
\end{dfn}

    что для любого $f$ существует отображение $h: T \To W$,
    что коммутативна диаграмма

\begin{dfn}
    $\phi : A \To B$ -- морфизм $k$-алгебр, $n \in \Z_{\ges 0}$, $S$ -- подмодуль
    $A$-модуля $A^n$. Обозначим $S^B := \gen{\phi(S)}_B$ $B$-линейную оболочку
    подмножества $\phi(S) \subseteq B^n$.
\end{dfn}

\begin{lem}\label{lem-isomorphism-b-modules}
    Пусть $\phi: \; A \To B$ -- морфизм $k$-алгебр, $n \in \Z_{\ges 0}$,
    $S$, $T$ -- подмодули $A$-модуля $A^n$, $A^n = S \oplus T$.
    Тогда выполняются следующие утверждения:
    \begin{enumerate}
        \item Существует такой изоморфизм $B$-модулей
        $$\psi : (B \otimes_{A} S) \oplus (B \otimes_{A} T) \To B^n,$$
        что для всех $b \in B$, $s \in S$, $t \in T$
        $$(b \otimes_{A} s, b \otimes_{A} t) \overset{\psi}{\mapsto} b(s + t);$$
        \item
        $$ \psi\left( \left(B \otimes_{A} S\right) \times \set{0} \right) = S^B, $$
        $$ \psi\left( \set{0} \times \left(B \otimes_{A} B\right)  \right) = T^B, $$
        $$ B^n = \left( A^n \right)^B = S^B \oplus T^B; $$
        \item Ранг проективного $B$-модуля $S^B$ равен рангу проективного $A$-модуля $S$. (\ref{rank-of-projective-module})
    \end{enumerate}
\end{lem}

\begin{dfn} %%
    Пусть $M$ -- $R$-модуль кольца $R$. Множество всех $\operatorname{End}_R(M)$ будем называть
    \textit{главной линейной группой}. Обозначается главная линейная группа $\GL_R(M)$.
\end{dfn}

\begin{dfn}
    Любой морфизм колец $\phi: k \To A$ будем называть $k$-алгеброй.
\end{dfn}

$A$ автоматически является $k$-модулем по операции умножения
$$
c \cdot a = \phi(c)a, \; c \in k
$$

\begin{dfn}
    Пусть $S$ -- мультипликативное подмножество кольца $R, M$ —
    $R$-модуль. Морфизм $R$-модулей $\lambda : M \To L$ и $R$-модуль $L$
    называются \textit{локализацией $M$ по $S$}, если выполняются следующие условия:
    \begin{enumerate}
        \item  $S_L \subseteq GL_R(L)$;
        \item для любого морфизма $R$-модулей $\lambda' : M \To L'$,
        для которого $S_L' \subseteq GL_R(L')$, существует и единственен
        такой морфизм $\phi : L \To L'$, что коммутативна диаграмма

    \end{enumerate}
\end{dfn}

\section{КАТЕГОРИИ И ФУНКТОРЫ}

\begin{dfn}
    Граф $G$ -- это любая упорядоченная четверка $(V,A, \operatorname{dom}, \operatorname{cod})$, где $V$, $A$ -- любые множества,
    $\operatorname{dom}$, $\operatorname{cod}: A \To V$ -- произвольные отображения.
    Элементы $V$ -- вершины графа $G$, элементы $A$ -- стрелки, или ребра графа $G$.
    Если $a \in A$, то $\operatorname{dom}(a)\in V$ называется
    началом, или областью определения $a$,
    $\operatorname{cod}(a) \in V$ -- концом, или областью значений $a$, или кообластью $a$.
\end{dfn}
\begin{example}\label{cat-graph}

    Граф $G$ состоит из множества вершин $V = \set{A,B,C,D}$,
    множества стрелок $A = \set{e,f,g,h}$,
    отображений $\operatorname{dom}, \operatorname{cod}: A \To V$,
    $\operatorname{cod}(e) = \operatorname{cod}(e) = A$, $\operatorname{dom}(f) = A$, $\operatorname{cod}(f) = B$, $\operatorname{dom}(g) = A$, $\operatorname{cod}(d) = B$, $\operatorname{dom}(h) = B$, $\operatorname{cod}(h) = C$.

\begin{dfn}
    Если $a,b$ -- вершины графа $G$, то множество стрелок графа $G$
    из $a$ в $b$ обозначим $G(a,b)$,
    или $\operatorname{Mor}_G(a,b)$,
    или $\operatorname{Mor}(a,b)$, если ясно, о каком графе идет речь.
\end{dfn}

\begin{example*}
    В графе $G$ из \ref{cat-graph} $G(A,A) = \set{e}$, $G(A,B) = \set{f,g}$, $G(A,C) = \0 $,
$G(C,D) = \0$ и так далее.

\end{example*}

Если стрелка $f$ имеет начало $a$ и конец $b$,
то мы пишем $f: a \To b$, или $a \overset{f}{\To} b$.
Если мы желаем подчеркнуть, что стрелка $f: a \To b$ относится к графу $G$,
то пишем $a \overset{f}{\underset{G}{\To}} b$
Это бывает нужным, если на одном и том же множестве $V$ определено
более одного графа с множеством вершин $V$.

\begin{dfn}
    Множество вершин графа $G$ обозначим $\operatorname{Ob}(G)$.
\end{dfn}

\begin{dfn}
    Пусть $G,H$ -- графы. Морфизмом $G \To H$ назовем любое
    отображение $\phi: V(G) \cup A(G) \To V(H) \cup A(H)$ такое,
    что $\phi(V(G)) \subseteq V(H)$ (т. е. вершины $G$ переходят в вершины $H$),
    $\phi(A(G)) \subseteq A(H)$ (т. е. ребра $G$ переходят в ребра $H$),
    и если $a,b \in V(G)$, $f \in \operatorname{Mor}_G(a,b)$,
    то $\phi(f) \in \operatorname{Mor}_H(\phi(a),\phi(b))$
    (т. е. из $f: a \To b$ следует $\phi(f): \phi(a) \To \phi(b))$.
\end{dfn}

\begin{dfn}\label{cat-def}
    \textit{Категория $\mathcal{C}$} -- это граф $G$, в котором для произвольных $A,B,C \in \operatorname{Ob}(\mathcal{C})$
    задано произведение $\operatorname{Mor}(B,C) \times \operatorname{Mor}(A,B) \To \operatorname{Mor}(A,C)$, причем выполняются следующие условия.
    \begin{enumerate}
        \item Для $A,B,C,D \in \operatorname{Ob}(\mathcal(C))$ и
        $f \in \operatorname{Mor}(A,B)$,
        $g \in \operatorname{Mor}(B,C)$,
        $h \in \operatorname{Mor}(C,D)$ выполняется закон ассоциативности
        $(hg)f = h(gf)$;
        \item Для каждого $A \in \operatorname{Ob}(\mathcal{C})$
        найдется $e \in \operatorname{Mor}(A,A)$ такой,
        что для любого $B \in \operatorname{Ob}(\mathcal{C})$ и
        $f \in \operatorname{Mor}(A,B)$, $g \in \operatorname{Mor}(B,A)$ выполняются равенства
        $fe = f$ и $eg = g$.
    \end{enumerate}
\end{dfn}

\begin{dfn}
    Элементы $\operatorname{Ob}(\mathcal{C})$ называются \textit{объектами категории $\mathcal{C}$}.
\end{dfn}

\begin{dfn}
    Элементы $\operatorname{Mor}(X,Y)$ называются \textit{морфизмами $X$ в $Y$},
    или морфизмами $X \To Y$.
\end{dfn}

Если $\alpha \in \operatorname{Mor}(X,Y)$ и ясно, о какой категории идет речь,
то мы пишем $\alpha: X \To Y$, или $X \overset{\alpha}{\To} Y$.
Морфизм $e: A \To A$ из аксиомы 2 определения \ref{cat-def} называется \textit{тождественным},
или \textit{единичным} морфизмом объекта $A$.

\begin{dfn}
    Множество всех морфизмов категории $\mathcal{C}$ обозначим $\operatorname{Mor}(\mathcal{C})$.
\end{dfn}

\begin{dfn}
    Пусть $\mathcal{A}, \mathcal{B}$ -- категории.
    Говорят, что $F$ -- функтор $\mathcal{A} \To \mathcal{B}$,
    если каждому $X \in \operatorname{Ob}(\mathcal{A})$ сопоставлен некоторый объект

\begin{thm}\cite[стр. 49]{Volochkov}\label{f-cat-to-univsets-cat}
    Пусть $I$ -- множество, $(F_i)_{i \in I}$ -- семейство функторов
    из категории $\mathcal{C}$ в категорию $\mathcal{U}$-множеств $\mathcal{S}$.
    Существует и единственен такой функтор $F: \mathcal{C} \To \mathcal{S}$, что
    \begin{enumerate}
        \item для каждого $X \in \operatorname{Ob}(\mathcal{C})$
        $F(X) = \prod\limits_{i\in I} F_i(X)$;
        \item для любого морфизма $\phi: X \To Y$ в $\mathcal{C}$
        $F(\phi) = \prod\limits_{i\in I} F_i(\phi)$.
    \end{enumerate}
\end{thm}

\section{СИСТЕМЫ УРАВНЕНИЙ И ФУНКТОРЫ}

\begin{dfn}\label{k-functor}
    Любой функтор $\mathbf{Alg_k} \To \mathbf{Set}$ -- $k$-функтор.
\end{dfn}

\begin{dfn}
    Пусть $F: \mathbf{Alg_k} \To \mathbf{Set}$ -- функтор забвения, $\Gamma$ -- множество.
    Аффинным пространством $A^\Gamma_k$ над $k$
    размерности $|\Gamma|$ назовем $k$-функтор $F^\Gamma$ (\ref{f-cat-to-univsets-cat}).
\end{dfn}

Очевидно, $A_k^\Gamma$ -- $k$-функтор, любой $k$-алгебре $A$ сопоставляющий $A^\Gamma$,
и произвольному морфизму $k$-алгебр $\phi: A \To B$ соотносящий отображение
$\phi^\Gamma: A^\Gamma \To B^\Gamma$.

Если $n \in \N$ и $\Gamma = \set{1, . . . , n}$ ($\Gamma = \0$ при $n = 0$),
то для любых $a_1, . . . , a_n$ мы считаем,
что $(\alpha_\gamma)_\gamma \in \Gamma = (a_1, . . . , a_n)$.
Таким образом, для произвольного множества $S$
мы очевидным образом отождествляем $S^\Gamma$ и $S^n$.
Для краткости мы пишем $A_k^n := A_k^\Gamma$,
и называем $k$-функтор $A_k^n$ аффинным $k$-пространством над $k$.
Если ясно, о каком $k$ идет речь, можно писать $A^n$.

\begin{dfn}%%
    Системой уравнений над $k$ называется тройка
    \begin{equation}\label{eq-sys}
        S = (P, (T_\gamma)_{\gamma \in \Gamma}, (f_\delta)_{\delta \in \Delta})
    \end{equation}

    где $P$ — $k$-алгебра многочленов, $(T_\gamma)_{\gamma \in \Gamma} =: T$ -- ее базис,
    и $(f_\delta)_{\delta \in \Delta})$ -- любое семейство многочленов из $P$.
\end{dfn}

\begin{dfn}
    Решением системы уравнений \ref{k-functor} в $k$-алгебре $A$ называется
    произвольное семейство $a \in A^\Gamma$ такое,
    что $f_\delta(a) = 0$ для всех $\delta \in \Delta$.
\end{dfn}
